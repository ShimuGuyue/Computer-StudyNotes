				else
				{
					if (i == 0)
						automaton[i].nexts[index] = 0;
					else
						automaton[i].nexts[index] = automaton[automaton[i - 1].fail].nexts[index];
				}
			}
		}
	}

	vector<int> match(string &text)
	{
		int n = text.length();
		int m = pattern.length();
		
		vector<int> indexs;
		
		int index = 0;
		for (int i = 0; i < n; ++i)
		{
			index = automaton[index].nexts[text[i] - 'a'];
			if (index == m)
				indexs.push_back(i - m + 1);
		}
		return indexs;
	}
};
